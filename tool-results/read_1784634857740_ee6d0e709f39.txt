    63→  return `${Date.now()}-${Math.random().toString(36).substring(2, 9)}`;

    66→/** Mask PII for logging per RULE #16. */
    67→function maskPII(text: string): string {
    68→  // Mask Aadhaar-like patterns (12+ digits)
    69→  let masked = text.replace(/\b\d{4}\d{4,8}\b/g, "XXXX-XXXX$&".slice(-4));
    70→  // Mask PAN-like patterns (5 letters + 4 digits + 1 letter)
    71→  masked = masked.replace(/\b[A-Z]{5}\d{4}[A-Z]\b/g, "XXXXX$&".slice(-5));
    72→  return masked;
    73→}
    74→
    75→/**
    76→ * Convert dot-path strings to the field descriptor format expected
    77→ * by extractFieldsFromMessage / buildExtractionPrompt.
    78→ *
    79→ * Looks up the actual field type, label, and enum options from PHASE_CONFIGS
    80→ * so that local extraction can handle ENUM, NUMBER, CURRENCY, and BOOLEAN
    81→ * fields correctly instead of falling through to the AI path for every field.
    82→ */
    83→function toFieldDescriptors(
    84→  dotPaths: string[]
    85→): Array<{
    86→  dotPath: string;
    87→  label: string;
    88→  type: "TEXT" | "NUMBER" | "ENUM" | "BOOLEAN" | "CURRENCY" | "DATE";
    89→  enumOptions?: string[];
    90→}> {
    91→  return dotPaths.map((dp) => {
    92→    /** Look up the actual FieldConfig from PHASE_CONFIGS by dot-path. */
    93→    let foundType: "TEXT" | "NUMBER" | "ENUM" | "BOOLEAN" | "CURRENCY" | "DATE" = "TEXT";
    94→    let foundLabel = dp.split(".").pop() ?? dp;
    95→    let foundEnumOptions: string[] | undefined;
    96→
    97→    for (const config of Object.values(PHASE_CONFIGS)) {
    98→      for (const group of config.fieldGroups) {
    99→        for (const field of group.fields) {
   100→          if (field.dotPath === dp) {
   101→            foundType = field.type;
   102→            foundLabel = field.label;
   103→            if (field.enumOptions) {
   104→              foundEnumOptions = field.enumOptions.map((o) => o.value);
   105→            }
   106→          }
   107→        }
   108→      }
   109→    }
   110→
   111→    return {
   112→      dotPath: dp,
   113→      label: foundLabel,
   114→      type: foundType,
   115→      enumOptions: foundEnumOptions,
   116→    };
   117→  });
   118→}

   147→  /** Start or resume the interview. Returns the first AI message. */
   148→  async startNewInterview(): Promise<ChatMessage> {
   149→    // Load the project profile into the store
   150→    await this.store.loadProject(this.projectId);
   151→
   152→    const profile = this.store.getProfile();
   153→    const phase = profile?.completion.currentPhase ?? "APPLICANT_DISCOVERY";
   154→
   155→    this.state.currentPhase = phase;
   156→    this.state.startedAt = new Date().toISOString();
   157→    this.state.lastActivityAt = new Date().toISOString();
   158→
   159→    // Determine if this is a fresh start or a resume
   160→    const isFresh =
   161→      !profile ||
   162→      profile.completion.interactionCount === 0;
   163→
   164→    let aiMessage: ChatMessage;
   165→
   166→    if (isFresh) {
   167→      // Fresh interview — get the entry message for the first phase
   168→      const entryMsg = getPhaseEntryMessage(phase);
   169→      aiMessage = this.createAssistantMessage(entryMsg, phase);
   170→    } else {
   171→      // Resume — use resume handler (2 args: profile, conversationHistory)
   172→      const resumeCtx = buildResumeContext(profile!, this.state.messages);
   173→      const resumeText = generateResumeMessage(resumeCtx);
   174→      aiMessage = this.createAssistantMessage(resumeText, phase);
   175→    }
   176→
   177→    // Initialize the AI provider message history with a system prompt
   178→    this.messageHistory = [
   179→      {
   180→        role: "system",
   181→        content: this.buildSystemPrompt(phase),
   182→      },
   183→    ];
   184→
   185→    this.state.messages.push(aiMessage);
   186→    this.emitAIMessage(aiMessage);
   187→    return aiMessage;
   188→  }

   190→  /** Process a user message and return the AI response. */
   191→  async processUserMessage(message: string): Promise<ChatMessage> {
   192→    // 1. Add user message to state
   193→    const userMsg = this.createUserMessage(message);
   194→    this.state.messages.push(userMsg);
   195→    this.state.interactionCount++;
   196→    this.state.lastActivityAt = new Date().toISOString();
   197→
   198→    // Push the user's actual message to the AI provider history so the AI
   199→    // can see what the user really said (not just the question context).
   200→    this.messageHistory.push({ role: "user", content: message });
   201→
   202→    // 2. Parse user intent
   203→    let intent: ParsedUserIntent;
   204→    if (isGreeting(message)) {
   205→      intent = {
   206→        type: "GREETING",
   207→        rawText: message,
   208→        confidence: 0.95,
   209→      };
   210→    } else {
   211→      intent = parseUserIntent(
   212→        message,
   213→        this.state.currentPhase,
   214→        this.state.messages.map((m) => ({
   215→          content: m.content,
   216→          role: m.role,
   217→        }))
   218→      );
   219→    }
   220→
   221→    // 3. Route based on intent type
   222→    let aiMessage: ChatMessage;
   223→
   224→    switch (intent.type) {
   225→      case "ANSWER":
   226→        aiMessage = await this.handleAnswer(message, intent);
   227→        break;
   228→      case "CORRECTION":
   229→        aiMessage = await this.handleCorrection(message, intent);
   230→        break;
   231→      case "CLARIFICATION":
   232→        aiMessage = await this.handleClarification(message);
   233→        break;
   234→      case "SKIP":
   235→        aiMessage = await this.handleSkip();
   236→        break;
   237→      case "GO_BACK":
   238→        aiMessage = await this.handleGoBack(intent);
   239→        break;
   240→      case "REVIEW_REQUEST":
   241→        aiMessage = await this.handleReviewRequest();
   242→        break;
   243→      case "CONFIRM":
   244→        aiMessage = await this.handleConfirm();
   245→        break;
   246→      case "HELP":
   247→        aiMessage = await this.handleHelp();
   248→        break;
   249→      case "GREETING":
   250→        aiMessage = await this.handleGreeting();
   251→        break;
   252→      case "OUT_OF_SCOPE":
   253→      default:
   254→        aiMessage = await this.handleOutOfScope(message);
   255→        break;
   256→    }
   257→
   258→    // 4. Add AI message to state and history
   259→    this.state.messages.push(aiMessage);
   260→    this.messageHistory.push({
   261→      role: "assistant",
   262→      content: aiMessage.content,
   263→    });
   264→
   265→    this.emitAIMessage(aiMessage);
   266→    return aiMessage;
   267→  }

   325→  private async handleAnswer(
   326→    message: string,
   327→    _intent: ParsedUserIntent
   328→  ): Promise<ChatMessage> {
   329→    const profile = this.store.getProfile();
   330→    if (!profile) {
   331→      return this.createAssistantMessage(
   332→        "Sorry, I couldn't load the project. Please try again.",
   333→        this.state.currentPhase
   334→      );
   335→    }
   336→
   337→    const phase = this.state.currentPhase;
   338→
   339→    // a. Get current question plan to know which fields we're collecting
   340→    const currentPlan = planNextQuestion(profile, phase, this.state.messages);
   341→    const targetFieldPaths = currentPlan.targetFields;
   342→
   343→    // b. Convert to field descriptors and extract locally
   344→    const fieldDescriptors = toFieldDescriptors(targetFieldPaths);
   345→    const localResult = extractFieldsFromMessage(
   346→      message,
   347→      fieldDescriptors,
   348→      "USER"
   349→    );
   350→
   351→    // c. Build extraction prompt and use AI for any remaining fields
   352→    const aiExtractions: FieldExtraction[] = [];
   353→    const fieldsNeedingAI = localResult.needsAI;
   354→
   355→    if (fieldsNeedingAI.length > 0) {
   356→      /**
   357→       * Use try/finally to guarantee the extraction prompt is always removed
   358→       * from messageHistory, even if getAIResponse or parseAIExtractionResponse
   359→       * throws an unexpected error. Without this, the extraction prompt would
   360→       * leak into the conversation history permanently.
   361→       */
   362→      try {
   363→        const aiFieldDescriptors = toFieldDescriptors(fieldsNeedingAI);
   364→        const extractionPrompt = buildExtractionPrompt(
   365→          message,
   366→          aiFieldDescriptors,
   367→          phase
   368→        );
   369→        this.messageHistory.push({ role: "user", content: extractionPrompt });
   370→        try {
   371→          const aiResp = await getAIResponse(this.messageHistory);
   372→          if (aiResp.success && aiResp.content) {
   373→            const parsed = parseAIExtractionResponse(aiResp.content, fieldsNeedingAI);
   374→            aiExtractions.push(...parsed);
   375→          }
   376→        } finally {
   377→          // Remove the extraction prompt from history (it was internal)
   378→          this.messageHistory.pop();
   379→        }
   380→      } catch (err) {
   381→        console.error("[InterviewController] AI extraction failed:", err);
   382→      }
   383→    }
   384→
   385→    // d. Merge all extractions
   386→    const allExtractions = [...localResult.extracted, ...aiExtractions];
   387→
   388→    // e. Update store for each extracted field
   389→    for (const extraction of allExtractions) {
   390→      this.store.updateField(
   391→        extraction.fieldPath,
   392→        extraction.value,
   393→        extraction.source
   394→      );
   395→    }
   396→
   397→    // f. After ACTIVITY_RESOLUTION phase: resolve NIC code via Knowledge Engine
   398→    if (phase === "ACTIVITY_RESOLUTION") {
   399→      await this.handleActivityResolution(profile, message);
   400→    }
   401→
   402→    // g. After activity is resolved, present Knowledge Engine suggestions
   403→    if (
   404→      phase === "PROJECT_SIZING" &&
   405→      profile.business.nicCode
   406→    ) {
   407→      await this.presentKnowledgeSuggestions(profile);
   408→    }
   409→
   410→    // h. Plan next question
   411→    const updatedProfile = this.store.getProfile()!;
   412→    const nextPlan = planNextQuestion(updatedProfile, phase, this.state.messages);
   413→
   414→    // i. Check if phase is complete, transition
   415→    if (nextPlan.completesPhase || nextPlan.isPhaseStart) {
   416→      const nextPhase = getNextPhase(phase, updatedProfile);
   417→      if (nextPhase && nextPhase !== phase) {
   418→        await this.transitionPhase(nextPhase);
   419→        // Get the entry message for the new phase
   420→        const entryMsg = getPhaseEntryMessage(nextPhase);
   421→        return this.createAssistantMessage(entryMsg, nextPhase);
   422→      }
   423→    }
   424→
   425→    // Generate the AI response for the next question
   426→    return this.generateAIResponse(nextPlan.question);
   427→  }

   429→  private async handleCorrection(
   430→    message: string,
   431→    intent: ParsedUserIntent
   432→  ): Promise<ChatMessage> {
   433→    const profile = this.store.getProfile();
   434→    if (!profile) {
   435→      return this.createAssistantMessage(
   436→        "No project loaded. Please try again.",
   437→        this.state.currentPhase
   438→      );
   439→    }
   440→
   441→    // The correction handler identifies fields from the message
   442→    const fields = intent.correctionFields ?? [];
   443→    if (fields.length === 0 && !intent.subject) {
   444→      return this.createAssistantMessage(
   445→        "I understand you want to correct something. Could you tell me which specific detail you'd like to change?",
   446→        this.state.currentPhase
   447→      );
   448→    }
   449→
   450→    // Extract fields and update
   451→    const extractionTargets = fields.length > 0 ? fields : [intent.subject ?? ""];
   452→    const fieldDescriptors = toFieldDescriptors(extractionTargets);
   453→    const result = extractFieldsFromMessage(message, fieldDescriptors, "USER");
   454→
   455→    for (const extraction of result.extracted) {
   456→      this.store.updateField(extraction.fieldPath, extraction.value, "USER");
   457→    }
   458→
   459→    if (result.extracted.length > 0) {
   460→      const fieldLabels = result.extracted.map((e) => e.label).join(", ");
   461→      const plan = planNextQuestion(this.store.getProfile()!, this.state.currentPhase, this.state.messages);
   462→      return this.generateAIResponse(
   463→        `Got it, I've updated ${fieldLabels}. ${plan.question}`
   464→      );
   465→    }
   466→
   467→    return this.createAssistantMessage(
   468→      "I couldn't identify what to correct. Could you be more specific? For example, 'Change my name to Rajesh Kumar'.",
   469→      this.state.currentPhase
   470→    );
   471→  }

   473→  private async handleClarification(message: string): Promise<ChatMessage> {
   474→    // Generate a helpful explanation about the current question
   475→    const plan = planNextQuestion(this.store.getProfile()!, this.state.currentPhase, this.state.messages);
   476→    const clarificationText = await this.generateAIResponse(
   477→      `The user asked for clarification about: "${message}". Explain clearly and simply what information is needed for: ${plan.question}. Keep it brief and in simple language.`
   478→    );
   479→    return clarificationText;
   480→  }

   482→  private async handleSkip(): Promise<ChatMessage> {
   483→    const updatedProfile = this.store.getProfile();
   484→    if (!updatedProfile) {
   485→      return this.createAssistantMessage(
   486→        "No project loaded.",
   487→        this.state.currentPhase
   488→      );
   489→    }
   490→
   491→    // Plan next question (which will skip the current field)
   492→    const nextPlan = planNextQuestion(updatedProfile, this.state.currentPhase, this.state.messages);
   493→
   494→    if (nextPlan.completesPhase || nextPlan.isPhaseStart) {
   495→      const nextPhase = getNextPhase(this.state.currentPhase, updatedProfile);
   496→      if (nextPhase && nextPhase !== this.state.currentPhase) {
   497→        await this.transitionPhase(nextPhase);
   498→        const entryMsg = getPhaseEntryMessage(nextPhase);
   499→        return this.createAssistantMessage(entryMsg, nextPhase);
   500→      }
   501→    }
   502→
   503→    return this.generateAIResponse(
   504→      `No problem, we'll skip that. ${nextPlan.question}`
   505→    );
   506→  }

   508→  private async handleGoBack(intent: ParsedUserIntent): Promise<ChatMessage> {

   510→    if (!targetPhase) {
   511→      return this.createAssistantMessage(
   512→        "Which section would you like to go back to? For example: applicant details, business details, or activity.",

   515→    }
   516→
   517→    await this.transitionPhase(targetPhase);
   518→    const plan = planNextQuestion(this.store.getProfile()!, targetPhase, this.state.messages);
   519→    return this.generateAIResponse(
   520→      `Sure, let's go back to ${PHASE_CONFIGS[targetPhase]?.label ?? targetPhase}. ${plan.question}`
   521→    );
   522→  }

   541→  private async handleConfirm(): Promise<ChatMessage> {
   542→    // Only valid in REVIEW phase
   543→    if (this.state.currentPhase !== "REVIEW") {
   544→      const plan = planNextQuestion(this.store.getProfile()!, this.state.currentPhase, this.state.messages);
   545→      return this.generateAIResponse(
   546→        `We're not at the review stage yet. ${plan.question}`
   547→      );
   548→    }
   549→
   550→    // Call store.confirmProject() which stamps all provenance CONFIRMED
   551→    this.store.confirmProject();
   552→
   553→    // Transition to VALIDATION_COMPLETION
   554→    await this.transitionPhase("VALIDATION_COMPLETION");
   555→
   556→    return this.createAssistantMessage(
   557→      "Your project has been confirmed! All details are now locked in. The system will now run validation and eligibility checks. You'll be notified once everything is ready for DPR generation.",
   558→      "VALIDATION_COMPLETION"
   559→    );
   560→  }

   608→  // ── Knowledge Engine Integration ──────────────────────────────────────────
   609→
   610→  private async handleActivityResolution(
   611→    _profile: ProjectProfile,
   612→    description: string
   613→  ): Promise<void> {
   614→    // Use Knowledge Engine to resolve activity — NEVER AI (RULE #4)
   615→    const suggestions = resolveActivity(description);
   616→
   617→    if (suggestions.length > 0) {
   618→      const bestMatch = suggestions[0];
   619→
   620→      // Check negative list
   621→      const negEntry = isOnNegativeList(bestMatch.nicCode);
   622→      const negKeyword = matchesNegativeKeyword(description);
   623→
   624→      if (negEntry || negKeyword) {
   625→        const reason = negEntry
   626→          ? `The activity "${bestMatch.description}" (NIC: ${negEntry.nicCode}) is on the PMEGP negative list: ${negEntry.reason}.`
   627→          : `Your activity description matches an excluded keyword: "${negKeyword}".`;
   628→
   629→        console.log(
   630→          `[InterviewController] Negative list match: ${maskPII(reason)}`
   631→        );
   632→        // The AI will communicate this to the user in the response
   633→      }
   634→
   635→      // Update the store with NIC code data (source: KNOWLEDGE per RULE #4)
   636→      this.store.updateField("business.nicCode", bestMatch.nicCode, "KNOWLEDGE");
   637→      this.store.updateField("business.nicDescription", bestMatch.description, "KNOWLEDGE");
   638→      this.store.updateField("business.sector", bestMatch.sector, "KNOWLEDGE");
   639→      this.store.updateField("business.subCategory", bestMatch.subCategory, "KNOWLEDGE");
   640→
   641→      // Emit suggestion event
   642→      this.emitSuggestion(
   643→        "business.nicCode",
   644→        bestMatch.nicCode,
   645→        `Resolved from activity description: ${bestMatch.matchReason}`
   646→      );
   647→    }
   648→  }

   650→  private async presentKnowledgeSuggestions(
   651→    profile: ProjectProfile
   652→  ): Promise<void> {
   653→    const nicCode = profile.business.nicCode;
   654→    if (!nicCode) return;
   655→
   656→    // Get machinery suggestions
   657→    const machinery = suggestMachinery(nicCode);
   658→    if (machinery.length > 0) {

   662→        `Knowledge Engine suggested ${machinery.length} machinery items for NIC ${nicCode}`
   663→      );
   664→    }
   665→
   666→    // Get raw material suggestions
   667→    const rawMaterials = suggestRawMaterials(nicCode);
   668→    if (rawMaterials.length > 0) {
   669→      this.emitSuggestion(
   670→        "rawMaterials.items",
   671→        rawMaterials,
   672→        `Knowledge Engine suggested ${rawMaterials.length} raw materials for NIC ${nicCode}`
   673→      );
   674→    }
   675→  }

   705→  // ── AI Response Generation ────────────────────────────────────────────────
   706→
   707→  private async generateAIResponse(
   708→    context: string,
   709→  ): Promise<ChatMessage> {
   710→    const phase = this.state.currentPhase;
   711→
   712→    // Add user context to message history
   713→    this.messageHistory.push({ role: "user", content: context });
   714→
   715→    // Ensure system prompt is present and up-to-date
   716→    if (this.messageHistory.length === 0 || this.messageHistory[0].role !== "system") {
   717→      this.messageHistory.unshift({
   718→        role: "system",
   719→        content: this.buildSystemPrompt(phase),
   720→      });
   721→    } else {
   722→      // Refresh system prompt with latest profile data
   723→      this.messageHistory[0] = {
   724→        role: "system",
   725→        content: this.buildSystemPrompt(phase),
   726→      };
   727→    }
   728→
   729→    try {
   730→      const response = await getAIResponse(this.messageHistory);
   731→
   732→      if (response.success && response.content) {
   733→        return this.createAssistantMessage(response.content, phase);
   734→      }
   735→
   736→      // AI failed — return a friendly fallback
   737→      console.error(
   738→        "[InterviewController] AI response failed:",
   739→        response.error
   740→      );
   741→      return this.createAssistantMessage(
   742→        "I'm having trouble connecting right now. Could you please repeat that?",
   743→        phase
   744→      );
   745→    } catch (err) {
   746→      console.error("[InterviewController] AI error:", err);
   747→      return this.createAssistantMessage(
   748→        "Something went wrong on my end. Please try again.",
   749→        phase
   750→      );
   751→    }
   752→  }
